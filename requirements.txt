# Requires Python 3.11  (mediapipe has NO wheels for 3.12+ on linux/aarch64)
#
# Raspberry Pi / DietPi (aarch64) — run setup_pi.sh instead of this directly.

#
# x86-64 Linux / Windows / macOS:
#   pip install -r requirements.txt
# ─────────────────────────────────────────────────────────────────────────────

# The aarch64 .whl is installed by setup_pi.sh; on x86-64 PyPI works fine.

websockets>=12.0,<14.0

# OS mouse cursor + click injection (gesture_engine.py Option A)
pynput>=1.7.0

# PyQt5 — overlay.py only; skip if you don't need the overlay on the Pi
# (apt install python3-pyqt5 is faster than pip on Debian/DietPi)
